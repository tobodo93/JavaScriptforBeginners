Let's look at an example of using the switch construct: 

var x = 3;         // declare a variable on which we will make a selection
var myResult = ""; // declare a variable into which we will write the result

switch (x) {
   case 1:         // if x is equal to one then write
     myResult = "X is equal to one!"; // in the variable myResult the phrase "X is equal to one!"
     break;
   case 2:
     myResult = "X is equal to two!";
     break;
   case 3:
     myResult = "X equals three!";
     break;
   default:
     myResult = "X is not 1, 2 or 3!";
}
document.write(myResult); // Output the result in a web document
